# File and Directory Management
basename : Strip directory and suffix from filenames.
cat : Display file content on screen.
cat > : Create a new file from standard input (Ctrl+D to save).
cat -n file1 : Number the lines in a specific file.
cd : Change current directory.
cd .. : Move up one directory level.
cd - : Switch to the previous directory.
cd ~ : Switch to the home directory.
cd ~user1 : Go to a specific user's home directory.
chmod : Change file mode bits (permissions).
chmod +x : Make a file executable.
chmod 755 : Set permissions to rwxr-xr-x (Owner: full, Group/Others: read+exec).
chmod 777 : Give full permissions to everyone (Dangerous).
chmod go-rwx directory1 : Remove read, write, and execute permissions for group and others on a directory.
chgrp : Change group ownership of files.
chown : Change file owner.
chown user:group : Change file owner and group simultaneously.
chattr : Change file attributes (e.g., make immutable: chattr +i file).
chattr +a file1 : Allow only appending to a file.
chattr +c file1 : Enable automatic compression/decompression by the kernel.
chattr +d file1 : Ensure the dump program ignores the file during backup.
chattr +s file1 : Allow a file to be deleted securely.
chattr +S file1 : Ensure modifications are written synchronously.
chattr +u file1 : Allow recovering file content after deletion.
cp : Copy a file.
cp -r : Copy a directory recursively.
cp dir/* . : Copy all files from a directory to the current working directory.
cp -a /tmp/dir1 . : Copy a directory preserving attributes to the current directory.
cp -a dir1 dir2 : Copy directory dir1 to dir2 preserving attributes.
df : Report file system disk space usage in 1K-blocks.
df -h : Report file system disk space usage in human-readable format (GB, MB).
df -T : Show file system type along with usage.
dirname : Strip last component from file name.
du : Estimate file space usage.
du -sh : Show summary of directory size in human-readable format.
du -sk * | sort -rn : Show size of files and directories sorted by size.
file : Determine file type description.
find : Search for files in a directory hierarchy.
find . -name "*.txt" : Find all text files in current directory.
find . -size +100M : Find files larger than 100MB.
find / -name file1 : Search for a file or directory named file1 starting from root.
find / -user user1 : Find files or directories belonging to a specific user.
find /home/user1 -name "*.bin" : Find files with .bin extension in a user's home directory.
find /usr/bin -type f -atime +100 : Find executables not used in the last 100 days.
find /usr/bin -type f -mtime -10 : Find files created or modified in the last 10 days.
find / -name ".rpm" -exec chmod 755 '{}' \; : Find .rpm files and change their permissions.
find / -xdev -name ".rpm" : Find .rpm files ignoring removable devices.
gunzip : Decompress files created by gzip.
gzip : Compress files using Lempel-Ziv coding.
gzip -9 file1 : Compress a file with maximum compression.
head : Output the first 10 lines of a file.
head -n : Output the first N lines of a file (e.g., head -n 20 file).
ln : Create hard or symbolic links.
ln -s : Create a symbolic link (shortcut) to a file or directory.
locate : Find files by name using a prebuilt database (instant search).
locate "*.ps" : Search for files with .ps extension.
ls : List directory contents (names only).
ls -a : List all files including hidden files (starting with dot).
ls -l : List files in long format (permissions, ownership, size, date).
ls -la : List all files (including hidden) in long format.
ls -lh : List files with sizes in human-readable format (KB, MB, GB).
ls -R : List directory contents recursively.
ls [0-9]* : List files and directories starting with numbers.
lsattr : List file attributes set with chattr.
lstree : Display files and directories in a tree format starting from root.
mkdir : Create a new directory (folder).
mkdir -p : Create nested directories (e.g., mkdir -p dir1/dir2).
mkdir dir1 dir2 : Create multiple directories simultaneously.
more : View file contents one screen at a time (legacy).
pwd : Print the absolute path of the current working directory.
realpath : Print the resolved absolute path.
rm : Remove a file.
rm -f : Force remove a file without confirmation prompts.
rm -r : Remove a directory and its contents recursively.
rm -rf : Force remove a directory and its contents recursively (Dangerous).
rmdir : Remove an empty directory.
shred : Securely delete a file by overwriting.
split : Split a large file into smaller parts.
stat : Display file or file system status.
tar : Archive utility to create and extract .tar files.
tar -cvf : Create a new uncompressed tar archive.
tar -czvf : Create a compressed gzip archive (.tar.gz).
tar -xvf : Extract a tar archive.
tar -xzvf : Extract a compressed gzip archive (.tar.gz).
tar -tf archive.tar : View contents of a tar archive.
tar -cvfj archive.tar.bz2 dir1 : Create a bzip2 compressed tar archive.
tar -xvfj archive.tar.bz2 : Extract a bzip2 compressed tar archive.
touch : Create an empty file or update file timestamps.
touch -t 0712250000 file1 : Modify the timestamp of a file or directory.
tree : List contents of directories in a tree-like format.
truncate : Shrink or extend the size of a file.
umask : Set file mode creation mask (default permissions).
unzip : Extract a zip archive.
wc : Print newline, word, and byte counts for each file.
wc -l : Count the number of lines in a file.
whereis : Locate the binary, source, and manual page files for a command.
which : Locate the executable path of a command.
zip : Create a zip archive.
zip -r : Zip a directory recursively.

egrep : Extended grep for complex regex patterns.
fgrep : Fixed string matching (fastest for literal strings).
fold : Wrap each input line to fit in specified width.
grep : Print lines matching a pattern.
grep -i : Search for a text string case-insensitively.
grep -r : Search recursively for text inside all files in a directory.
grep -v : Invert match (show lines that do NOT match the pattern).
history : Display the command history list.
history -c : Clear the command history.
join : Join lines of two files on a common field.
paste : Merge lines of files.

rev : Reverse lines of a file or string.
sdiff file1 file2 : Interactively find differences and merge files.
sed : Stream editor for filtering and transforming text.
sed 's/old/new/g' : Replace all occurrences of 'old' with 'new' in stream.
sed '/^$/d' example.txt : Delete all empty lines from a file.
sed '/^#/d; /^$/d' example.txt : Delete comments and empty lines from a file.
sed -e '1d' example.txt : Remove the first line from a file.
sed -e 's/ *$//' example.txt : Remove trailing whitespace from each line.
sed -e 's/00/0/g' example.txt : Replace multiple zeros with a single zero.
sed -e 's/string1//g' example.txt : Remove a specific string from text.
sed -n '/string1/p' : Display lines containing a specific string.
sed -n '1,5p' example.txt : Print lines 1 to 5 from a file.
sed -n '5p;5q' example.txt : Display line 5 of a file.
sort : Sort lines of text files.
sort -n : Sort lines numerically.
sort -r : Sort lines in reverse order.
sort file1 file2 | uniq : Sort and remove duplicate lines from two files.
sort file1 file2 | uniq -d : Sort and display only duplicate lines from two files.
sort file1 file2 | uniq -u : Sort and display only unique lines from two files.
strings : Print the sequences of printable characters in files.
tee : Write output to both standard output and a file.
tr : Translate or delete characters.
uniq : Remove adjacent duplicate lines.
unix2dos fileunix.txt filedos.txt : Convert a text file from Unix to MSDOS format.
xargs : Build and execute command lines from standard input.
